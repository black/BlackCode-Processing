Name image()
Examples 
import hypermedia.video.*;


OpenCV opencv;

void setup() {
    size( 320, 240 );
    opencv = new OpenCV( this );
    opencv.capture( width, height );  // open video stream
}

void draw() {
    opencv.read();                   // grab frame from camera
    image( opencv.image(), 0, 0 );   // and display image
}
