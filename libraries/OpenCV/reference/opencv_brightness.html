Name brightness()
Examples 
import hypermedia.video.*;

OpenCV opencv;

void setup() {

    size( 320, 240 );

    opencv = new OpenCV(this);
    opencv.capture( width, height );
}

void draw() {

    opencv.read();
    opencv.convert( GRAY );
    opencv.ROI( 110, 70, 100, 100 );
    opencv.brightness( 80 );
    opencv.contrast( 90 );

    image( opencv.image(), 0, 0 );
}
